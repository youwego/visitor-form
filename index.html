  <script>

    function submitForm() {

      const artist = document.getElementById("artist").value.trim();

      const presale = document.getElementById("presale").value.trim();

      const guest = document.getElementById("guest").value.trim();

      const memo = document.getElementById("memo").value.trim();


      if (!artist || !presale || !guest) {

        alert("すべての必須項目を入力してください");

        return;

      }


      const params = new URLSearchParams({

        artist: artist,

        presale: presale,

        guest: guest,

        memo: memo

      });


      const endpoint = "https://script.google.com/macros/s/AKfycbzmkSB11hLhbevlqPblXZ4e4WnDxPSp7wSUpoHbuxdQUjmz2WdPVoJ8z3M4dEo8yrgu/exec";


      console.log("✅ submit URL:", `${endpoint}?${params.toString()}`);


      fetch(`${endpoint}?${params.toString()}`, {

        method: "GET",

      })

      .then(() => {

        alert("送信しました。スプレッドシートに反映されます。");

        document.getElementById("form").reset();

      })

      .catch((error) => {

        alert("送信中にエラーが発生しました: " + error.message);

      });

    }

  </script>
